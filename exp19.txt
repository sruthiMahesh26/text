name: Deploy Site

on:
  push:
    branches:
      - main

permissions:
  contents: write   # ✅ IMPORTANT (fixes git error)

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4   # ✅ updated version

      - name: Deploy
        uses: peaceiris/actions-gh-pages@v4   # ✅ updated version
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./dist
